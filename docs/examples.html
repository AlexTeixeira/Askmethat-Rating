Basic Examples

Simple select rating

Simple clickable rating

HTML
              
                <div id="myBasicRating"></div>
JavaScript
              
                var options = {
                    backgroundColor : "#e5e500",
                    hoverColor: "#ffff66",
                    fontClass: "fa fa-star",
                    minRating: 1,
                    maxRating: 5,
                    readonly: false,
                    step: AskmethatRatingSteps["OnePerOneStep"]                                       
                    };

                var div = document.getElementById("myBasicRating");
                var amtRating = new AskmethatRating(div, 2, options);

JavaScript
              
                  var decimalOpt = {
                    backgroundColor : "#e5e500",
                    hoverColor: "#ffff66",
                    fontClass: "fa fa-star",
                    minRating: 1,
                    maxRating: 5,
                    readonly: false,
                    step: AskmethatRatingSteps["DecimalStep"]
                  };

                  var halpOpt = {
                    backgroundColor : "#e5e500",
                    hoverColor: "#ffff66",
                    fontClass: "fa fa-star",
                    minRating: 1,
                    maxRating: 5,
                    readonly: false,
                    step: AskmethatRatingSteps["HalfStep"]
                  };

                  var FullOpt = {
                    backgroundColor : "#e5e500",
                    hoverColor: "#ffff66",
                    fontClass: "fa fa-star",

                  var div = document.getElementById("fullStepRating");

Working with events

On change event

Event fired when the value of the rating change

HTML
              
                <div id="changeEventRating"></div>
                <span id="changeEventRatingValue">0<span>
JavaScript
              
                var options = {
                    backgroundColor : "#e5e500",
                    hoverColor: "#ffff66",
                    fontClass: "fa fa-star",
                    minRating: 0,
                    maxRating: 5,
                    readonly: false,
                    step: AskmethatRatingSteps["HalfStep"]                                       
                    };

                var changeDiv = document.getElementById("changeEventRating");
                var amtRating = new AskmethatRating(div, 2, options);

                 changeDiv.addEventListener("amt-change", function(e){
                  document.getElementById("changeEventRatingValue").innerHTML = e.detail;
                });
